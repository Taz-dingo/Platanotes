## 深度和高度

[543. 二叉树的直径](https://leetcode.cn/problems/diameter-of-binary-tree/)

```typescript
function diameterOfBinaryTree(root: TreeNode | null): number {
    let maxDiameter = 0;
    function depth(node){
        if(!node) return 0;

        const LD = depth(node.left);
        const RD = depth(node.right);

        maxDiameter = Math.max(maxDiameter,LD+RD);

        return Math.max(LD,RD)+1;
    }
    depth(root);

    return maxDiameter;
};
```



```js
// 深度：当前节点到叶子节点。通过“递”阶段，传参接力
function getNodeDepth(root) {
    const res = [];

    (function dfs(node, depth) {
        if (!node) return;

        res.push({ val: node.val, depth });

        const LD = dfs(node.left, depth + 1);
        const RD = dfs(node.right, depth + 1);
    })(root, 0)

    return res;
}

// 高度：叶节点到当前节点。通过“归”阶段，return接力
function getNodeHeight(root) {
    const res = [];

    (function dfs(node) {
        if (!node) return 0;

        const LH = dfs(node.left);
        const RH = dfs(node.right);

        const height = Math.max(LH, RH);
        res.push({ val: node.val, height });

        return height + 1;
    })(root)

    return res;
}
```
