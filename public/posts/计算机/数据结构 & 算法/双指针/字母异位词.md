[438. 找到字符串中所有字母异位词](https://leetcode.cn/problems/find-all-anagrams-in-a-string/)

```typescript
function findAnagrams(s: string, p: string): number[] {
    const res = [];
    const sCount = new Array(26).fill(0);  
    const pCount = new Array(26).fill(0);  
    const charIndex = (char)=>(char.charCodeAt(0) - 'a'.charCodeAt(0));
	// 获取p的哈希表
    for(const char of p){
        pCount[charIndex(char)]++;
    }

    for(let i = 0;i < s.length;i++){
        sCount[ charIndex(s[i]) ]++; // 右边界右移

        // 左边界右移
        // 长度到位后动态移动
        if(i >= p.length){
            sCount[ charIndex(s[i - p.length]) ] --;
        }

        // 长度
        if(i >= p.length - 1 && sCount.join("") === pCount.join("")){
            res.push(i - p.length + 1);
        }
    }

    return res;
};
```

```js
// 这个会超时
function findAnagrams(s: string, p: string): number[] {
    const res = [];

    for(let i = 0;i < s.length - p.length + 1;i++){
        const curStr = s.slice(i,i+p.length);
        if(compareStr(curStr,p)){
            res.push(i);
        }
    }

    return res;
};

function compareStr(str1,str2){
    return str1.split("").sort().join("") === str2.split("").sort().join("");
}
```
