[455. 分发饼干](https://leetcode.cn/problems/assign-cookies/) - 先尝试满足大胃口

```typescript
function findContentChildren(g: number[], s: number[]): number {
    g.sort((a, b) => (a - b));  // 孩子
    s.sort((a, b) => (a - b));  // 饼干
    let count = 0;
    let curS = s.length - 1;
    let curG = g.length - 1;

    while(curG >= 0){
        if(s[curS] >= g[curG]){
            count++;
            curG--;
            curS--;
        }else{
            curG--;
        }
    }

    return count;
};
```
