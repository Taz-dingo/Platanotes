function groupAnagrams(strs: string[]): string[][] {
    const map = new Map();

    for(let i = 0;i < strs.length;i++){
        const key = strs[i].split("").sort().join("");

        if(!map.has(key)){
            map.set(key,[strs[i]]);
        }else{
            map.get(key).push(strs[i]);
        }
    }

    return Array.from(map.values());
};
